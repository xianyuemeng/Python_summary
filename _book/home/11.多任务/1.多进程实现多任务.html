  • 多进程实现多任务

multiprocessing 库

跨平台版本的多进程模块，提供了一个Process类来代表一个进程对象

  • 进程

    • 对于操作系统而言，一个任务就是一个进程
    • 进程是系统中程序执行和资源分配的基本单位。每个进程都有自己的数据段、代码段和堆栈段
from multiprocessing import Process
import os
import time

def func(num):
    # os.getpid()获取当前id进程号
    # os.getppid()获取当前id进程的父进程id号
    print("子进程启动... -- %s -- %s -- %s" % (os.getpid(), os.getppid(), time.time()))
    time.sleep(3)
    print("子进程结束...", num)

if __name__ == '__main__':
    print("父进程启动... -- %s -- %s -- %s" % (os.getpid(), os.getppid(), time.time()))

    p = Process(target=func, args=(10,))
    p.start()

    # 让父进程等待子进程结束之后父进程再结束
    p.join()

    print("父进程结束...")

  • 全局变量在多个进程中不共享

from multiprocessing import Process
from time import sleep

num = 100

def run():
    print("子进程启动...")
    global num
    num += 1
    print("子进程结束...", num)

if __name__ == '__main__':
    print("父进程开始...")
    p = Process(target=run)
    p.start()
    p.join()
    # 在子进程中修改全局变量对父进程中的全局变量并没有影响
    # 在创建子进程时对全局变量做了一个备份
    print("父进程结束...", num)

  • 进程池

# -*- coding:utf-8 -*-

from multiprocessing import Pool
import os, time, random

def run(name):
    print("子进程%d启动... -- %s" % (name, os.getpid()))
    start = time.time()
    time.sleep(random.choice([1, 2, 3]))
    end = time.time()
    print("子进程%d结束... -- %s，耗时%.2f" % (name, os.getpid(), end-start))

if __name__ == '__main__':
    print("父进程开始...")
    # 创建多个进程
    # Pool 进程池
    # 进程池默认是CPU核心数
    pp = Pool()
    for i in range(10):
        # 创建进程，放入进程池统一管理
        pp.apply_async(run, args=(i,))
    # 在调用join之前，必须先调用close，而且调用close之后就不能再继续添加新的进程了
    pp.close()
    pp.join()
    print("父进程结束...")
