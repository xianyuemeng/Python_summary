# -*- coding:utf-8 -*-

import os
import pickle
import random

from Bank.Bank05.person import Person

from Bank.Bank05.card import Card


class Operation:
    # 初始化加载用户信息
    def __init__(self):
        self.get_user_dict()
        print(self.user_dict)

    # 获取用户信息
    def get_user_dict(self):
        if os.path.exists("user.txt"):
            with open("user.txt", "rb") as f:
                self.user_dict = pickle.load(f)
        else:
            self.user_dict = {}

    # 退出银行
    def exit_bank(self):
        with open("user.txt", "wb") as f:
            pickle.dump(self.user_dict, f)

    # 开户
    def register(self):
        name = input("请输入您的姓名：")
        pwd = input("请输入您的密码：")
        pwd1 = input("请再次输入您的密码：")
        if pwd == pwd1:
            pwd2 = pwd
        else:
            print("您两次输入密码不一致！开户失败...")
            return -1
        phone = input("请输入您的手机号码：")
        id_card = input("请输入您的身份证：")
        card_id = self.get_card_id()
        # 实例化卡和人
        card = Card(card_id, pwd2, 10)
        user = Person(name, id_card, card)
        # 将注册的信息存入字典
        self.user_dict[card_id] = user
        print("恭喜您，%s，注册成功！您的卡号为：%s，卡内余额：%s。（请牢记您的卡号！）" % (user.name, card.card_id, card.money))

    # 查询
    def query(self):
        user = self.get_user_info()
        if not user:
            print("卡号不存在！查询失败...")
        else:
            if user.card.islock:
                print("卡号已被锁定！查询失败...")
            else:
                check = self.check_password(user)
                if check:
                    print("卡内余额：%s" % (user.card.money))
                else:
                    user.card.islock = True
                    print("密码连续输入错误3次，卡已被锁定！")

    # 存钱
    def save_money(self):
        user = self.get_user_info()
        if not user:
            print("卡号不存在！存钱失败...")
        else:
            if user.card.islock:
                print("卡号已被锁定！存钱失败...")
            else:
                check = self.check_password(user)
                if check:
                    new_money = int(input("请输入您想存入的金额："))
                    if new_money < 0:
                        print("您输入的金额错误，存钱失败...")
                    else:
                        user.card.money += new_money
                        print("恭喜您，存钱成功！卡内余额：%s." % (user.card.money))
                else:
                    user.card.islock = True
                    print("密码连续输入错误3次，存钱失败，卡已被锁定！")

    # 取钱
    def get_money(self):
        user = self.get_user_info()
        if not user:
            print("卡号不存在！取款失败...")
        else:
            if user.card.islock:
                print("卡号已被锁定！取款失败...")
            else:
                check = self.check_password(user)
                if check:
                    new_money = int(input("请输入您想取出的金额："))
                    if 0 < new_money < user.card.money:
                        user.card.money -= new_money
                        print("恭喜您，取款成功！卡内余额：%s." % (user.card.money))
                    else:
                        print("您输入的金额错误，取款失败...")
                else:
                    user.card.islock = True
                    print("密码连续输入错误3次，取款失败，卡已被锁定！")

    # 转账
    def transfer_money(self):
        card_id1 = input("请输入您想转入的卡号：")
        user1 = self.user_dict[card_id1]
        if not user1:
            print("转入卡号不存在！转账失败...")
            return -1
        if user1.card.islock:
            print("转入卡号已被锁定！转账失败...")
            return -1

        card_id = input("请输入您的卡号：")
        user = self.user_dict[card_id]
        if not user:
            print("您的卡号不存在！转账失败...")
            return -1
        if user.card.islock:
            print("你的卡号已被锁定！转账失败...")
            return -1

        check = self.check_password(user)
        if not check:
            print("您输入密码连续错误3次，转账失败，卡已被锁定！")
        else:
            new_money = int(input("请输入您想要转出的金额："))
            if new_money < 0:
                print("您输入金额有误，转账失败！")
                return -1
            if new_money > user.card.money:
                print("您卡内余额不足，转账失败！")
                return -1
            user.card.money -= new_money
            user1.card.money += new_money
            print("恭喜您！转账成功...此次转账金额：%s" % (new_money))


    # 改密
    def change_pwd(self):
        user = self.get_user_info()
        if not user:
            print("卡号不存在！改密失败...")
        else:
            if user.card.islock:
                print("卡号已被锁定！改密失败...")
            else:
                check = self.check_password(user)
                if check:
                    new_password = input("请输入新密码：")
                    new_password1 = input("请再次输入新密码：")
                    if new_password == new_password1:
                        user.card.password = new_password
                        print("恭喜您，改密成功！")
                    else:
                        print("两次输入密码不一致，改密失败！")
                else:
                    user.card.islock = True
                    print("密码连续输入错误3次，改密失败，卡已被锁定！")

    # 挂失
    def lock(self):
        user = self.get_user_info()
        if not user:
            print("卡号不存在！挂失失败...")
        else:
            id_card = input("请输入您的身份证号：")
            if id_card == user.id_card:
                user.card.islock = True
                print("恭喜您，挂失成功！")
            else:
                print("身份证输入错误！挂失失败！")

    # 解锁
    def unlock(self):
        user = self.get_user_info()
        if not user:
            print("卡号不存在！解锁失败...")
        else:
            if not user.card.islock:
                print("该卡未被锁定，无需解锁！")
            else:
                check = self.check_password(user)
                if not check:
                    print("密码错误三次，解锁失败！")
                else:
                    user.card.islock = False
                    print("恭喜您，解锁成功！")

    # 随机获取5位卡号（且不能重复）
    def get_card_id(self):
        while True:
            card_id = str(random.randint(10000, 99999))
            if card_id not in self.user_dict:
                return card_id

    # 获得一个用户
    def get_user_info(self):
        card_id = input("请输入您的卡号：")
        if card_id not in self.user_dict:
            return False
        else:
            user = self.user_dict.get(card_id)
            return user

    # 检查输入密码，错3次锁定该卡
    def check_password(self, user):
        count = 1
        while count < 4:
            pwd = input("请输入您的密码：")
            if pwd == user.card.password:
                return True
            count += 1
        return False

      # -*- coding:utf-8 -*-

  import time

  from Bank.Bank05.operation import Operation
  from Bank.Bank05.view import View

  v = View()

  def main():
      while True:
          if v.login():
              v.login_welcome()
              print('登陆成功！请稍后...')
              for i in range(3):
                  print('*' * (3 - i))
                  time.sleep(1)
              o = Operation()
              break
          else:
              print("账号或密码输入错误！")
      while True:
          try:
              v.login_operation()
              choice = input("请选择您想要进行的服务：")
              # 开户
              if choice == '1':
                  o.register()

              # 查询
              elif choice == '2':
                  o.query()

              # 存钱
              elif choice == '3':
                  o.save_money()

              # 取钱
              elif choice == '4':
                  o.get_money()

              # 转账
              elif choice == '5':
                  o.transfer_money()

              # 改密
              elif choice == '6':
                  o.change_pwd()

              # 挂失
              elif choice == '7':
                  o.lock()

              # 解锁
              elif choice == '8':
                  o.unlock()

              # 退出
              elif choice == '9':
                  o.exit_bank()
                  print("正在退出，请稍后...")
                  time.sleep(1)
                  print("已退出！")
                  break

              else:
                  raise Exception("您输入的序号有误，请重新输入！")
              # 每执行一次就停顿0.5s
              time.sleep(0.5)

          except BaseException as e:
              print(e)
  if __name__ == '__main__':
      main()
