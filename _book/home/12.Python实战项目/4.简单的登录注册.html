# coding:utf-8

import pymysql

# 连接数据库
connect = pymysql.connect(
    host = "localhost",
    user = "root",
    password = "123456",
    port = 3306,
    db="xdl",
    charset = "utf8",
        )
# 实例化游标
cur = connect.cursor()
while True:
    user_name = input("请输入您的用户名：")
    sql = "select * from user"
    cur.execute(sql)
    results = cur.fetchall()
    for row in results:
        name = row[0]
        if user_name == name:
            print("该用户名已被注册，请重新输入！")
            break
    else:
        while True:
            user_pwd = input("请输入您的密码：")
            user_pwd2 = input("请再次输入您的密码：")
            if user_pwd == user_pwd2:
                print("恭喜您，注册成功！")
                sql = "insert into user values('%s','%s') "
                date = (user_name, user_pwd)
                cur.execute(sql %date)
                connect.commit()
                cur.close()
                connect.close()
                break
            else:
                print("您两次输入的密码不一样，请重新输入！")
        break

# coding:utf-8
import sys
import pymysql

# 连接数据库
connect = pymysql.connect(
    host="localhost",
    user="root",
    password="123456",
    port=3306,
    db="xdl",
    charset="utf8",
    )
i = 0
# 实例化游标
cur = connect.cursor()

while i < 3:
    b = 1
    while b != 0:
        # 获取用户名
        account_name = input('请输入您的用户名：')
        # 读取数据库中被锁定用户名的列表，并和获取的用户名进行比较，若存在相同的，则判定该用户名已被锁定
        cur.execute("select * from user_lock")
        results_lock = cur.fetchall()
        # print(results_lock)
        if results_lock == ():
            break
        lock_name = []
        for lock_n in results_lock:
            lock_name.append(lock_n)
            # print(lock_name)
        # print((account_name,))
        if (account_name,) in lock_name:
            print("该用户名已被锁定，请重新输入！")

        else:
            b = 0

    # 读取数据库中所有用户名和密码
    cur.execute("select * from user")
    results = cur.fetchall()
    for user_n in results:
        # 读取数据库中所有用户名
        user_name = user_n[0]
        # 读取数据库中所有用户名对于的密码
        user_pwd = user_n[1]
        if user_name == account_name:
            j = 0
            while j < 3:
                # 获取用户输入的用户名相应的密码
                account_pwd = str(input('请输入您的密码：'))
                if account_pwd == user_pwd:
                    print('%s,恭喜您登陆成功！' % account_name)
                    cur.close()
                    connect.close()
                    sys.exit()
                else:
                    if j < 3:
                        print('警告：%s 输入密码错误，请重新输入！您还有 %d 次机会。' % (account_name,2-j))
                j += 1
            else:
                sql = "insert into user_lock value('%s')"
                date = account_name
                cur.execute(sql % date)
                connect.commit()
                cur.close()
                connect.close()
                print('警告：%s 您已达到最大登陆次数，账号将被锁定！' % account_name)
                sys.exit()
    else:
        print('警告：%s 不存在，请重新输入！' % account_name)

# coding:utf-8

import sys
import pymysql
# 连接数据库
connect = pymysql.connect(
    host = "localhost",
    user = "root",
    password = "123456",
    port = 3306,
    db = "xdl",
    charset = "utf8",
)

# 实例化游标
cur = connect.cursor()

unlock_name = input("请输入您想解锁的用户名：")

sql = "select * from user_lock"
cur.execute(sql)
result_lock = cur.fetchall()
for lock_n in result_lock:
    lock_name = lock_n[0]
    if lock_name == unlock_name:
        sql = "delete from user_lock where name='%s'"
        date = (unlock_name)
        cur.execute(sql %date)
        connect.commit()
        cur.close()
        connect.close()
        print("恭喜您，用户 %s 已经成功解锁！" %unlock_name)
