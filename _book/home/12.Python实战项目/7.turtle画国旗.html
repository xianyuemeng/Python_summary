#coding:utf-8
import turtle
#封装移动函数
def My_goto(x,y):
    turtle.penup()
    turtle.goto(x,y)
    turtle.pendown()
#定义矩形函数
def Draw_four(size1,size2,angle,x,y,color='red'):
    My_goto(x, y)
    turtle.pencolor(color)
    turtle.fillcolor(color)
    turtle.begin_fill()
    for i in range(2):
        turtle.forward(size1)
        turtle.left(angle)
        turtle.forward(size2)
        turtle.left(angle)
    turtle.end_fill()
#定义五角星函数
def Draw_5_angle(size,color='yellow'):
    turtle.pencolor(color)
    turtle.fillcolor(color)
    turtle.begin_fill()
    for i in range(5):
        turtle.forward(size)
        turtle.right(144)
    turtle.end_fill()

turtle.speed(10) #画笔速度
turtle.pensize(0) #画笔粗细
#先画一个矩形
Draw_four(600,400,90,-300,-200)
#在左上角画一个大五角星
My_goto(-250,95)
Draw_5_angle(114)
#画4个小五角星
a = [320,25,15,15]
b = [0,1,1,0]
c = [140,100,40,0]
for i in range(4):
    My_goto((-120+b[i]*30),(c[i]))
    turtle.right(a[i])
    Draw_5_angle(38)

turtle.hideturtle() #隐藏小海龟
turtle.mainloop()
